CREATE TABLE Region (ID_Region int, Naziv int, VremenskaZona int,  PRIMARY KEY ( ID_Region ));
CREATE TABLE Igrica (ID_Igrica int, Naziv int,  PRIMARY KEY ( ID_Igrica ));
CREATE TABLE StanjeAktivnosti (ID_Aktiv int, Vrednost int,  PRIMARY KEY ( ID_Aktiv ));
CREATE TABLE Grad (ID_grad int, Naziv int, BrojStanovnika int,  PRIMARY KEY ( ID_grad ));
CREATE TABLE Server (ID_Server int, BoostNivo int, BrojKorisnika int, Naziv int, URL int, ID_Region int,  PRIMARY KEY ( ID_Server ) , FOREIGN KEY (ID_Region) REFERENCES Region(ID_Region));
CREATE TABLE Kategorija (ID_Kategorija int, Naziv int, ID_Server int,  PRIMARY KEY ( ID_Kategorija ) , FOREIGN KEY (ID_Server) REFERENCES Server(ID_Server));
CREATE TABLE Kanal (ID_Kanal int, Naziv int, MxKorisnika int, ListaPoziva int, ID_Kategorija int,  PRIMARY KEY ( ID_Kanal ) , FOREIGN KEY (ID_Kategorija) REFERENCES Kategorija(ID_Kategorija));
CREATE TABLE _tekst (Brzina int, tema int, ID_Kanal int,  PRIMARY KEY ( ID_Kanal ) , FOREIGN KEY (ID_Kanal) REFERENCES Kanal(ID_Kanal));
CREATE TABLE _glas (Protok int, ID_Kanal int,  PRIMARY KEY ( ID_Kanal ) , FOREIGN KEY (ID_Kanal) REFERENCES Kanal(ID_Kanal));
CREATE TABLE Pozivnica (ID_Poziv int, VremeTrajanja int, URL int, ID_Server int,  PRIMARY KEY ( ID_Poziv ) , FOREIGN KEY (ID_Server) REFERENCES Server(ID_Server));
CREATE TABLE Dozvola (ID_Dozvola int, Naziv int, Pozicija int, Podrazumevana int, Boja int, Prikaz int, ID_Server int,  PRIMARY KEY ( ID_Dozvola ) , FOREIGN KEY (ID_Server) REFERENCES Server(ID_Server));
CREATE TABLE _admin (Tip int, ID_Dozvola int,  PRIMARY KEY ( ID_Dozvola ) , FOREIGN KEY (ID_Dozvola) REFERENCES Dozvola(ID_Dozvola));
CREATE TABLE _osnovna (Tema int, ID_Dozvola int,  PRIMARY KEY ( ID_Dozvola ) , FOREIGN KEY (ID_Dozvola) REFERENCES Dozvola(ID_Dozvola));
CREATE TABLE Poziv (ID_Poziv int, Trajanje int, Kvalitet int, ID_Kanal int,  PRIMARY KEY ( ID_Poziv ) , FOREIGN KEY (ID_Kanal) REFERENCES _glas(ID_Kanal));
CREATE TABLE Video_snimak (ID_Snimak int, Kvalitet int, Trajanje int, ID_Poziv int,  PRIMARY KEY ( ID_Snimak ) , FOREIGN KEY (ID_Poziv) REFERENCES Poziv(ID_Poziv));
CREATE TABLE Institucija (ID_Institucija int, Naziv int, ID_grad int,  PRIMARY KEY ( ID_Institucija ) , FOREIGN KEY (ID_grad) REFERENCES Grad(ID_grad));
CREATE TABLE Kurs (ID_Kurs int, Naziv int, ImePrezimeProf int, ESPB int, Semestar int, ID_Institucija int,  PRIMARY KEY ( ID_Kurs ) , FOREIGN KEY (ID_Institucija) REFERENCES Institucija(ID_Institucija));
CREATE TABLE Projekat (ID_Projekat int, Naziv int, Trajanje int, MxUcesnika int, ID_Kurs int,  PRIMARY KEY ( ID_Projekat ) , FOREIGN KEY (ID_Kurs) REFERENCES Kurs(ID_Kurs));
CREATE TABLE Korisnik (ID_Koris int, Diskriminator int, Username int, Nickname int, ID_Aktiv int, ID_Igrica int, ID_Poziv int,  PRIMARY KEY ( ID_Koris , Diskriminator ) , FOREIGN KEY (ID_Aktiv) REFERENCES StanjeAktivnosti(ID_Aktiv) , FOREIGN KEY (ID_Igrica) REFERENCES Igrica(ID_Igrica) , FOREIGN KEY (ID_Poziv) REFERENCES Poziv(ID_Poziv));
CREATE TABLE Ucenik (Razred int, ID_Koris int, Diskriminator int, ID_Institucija int,  PRIMARY KEY ( ID_Koris , Diskriminator ) , FOREIGN KEY (ID_Koris,Diskriminator) REFERENCES Korisnik(ID_Koris,Diskriminator) , FOREIGN KEY (ID_Institucija) REFERENCES Institucija(ID_Institucija));
CREATE TABLE Student (Godina int, ID_Koris int, Diskriminator int, ID_Institucija int,  PRIMARY KEY ( ID_Koris , Diskriminator ) , FOREIGN KEY (ID_Koris,Diskriminator) REFERENCES Korisnik(ID_Koris,Diskriminator) , FOREIGN KEY (ID_Institucija) REFERENCES Institucija(ID_Institucija));
CREATE TABLE Srednja_Skola (Tip int, ID_Institucija int,  PRIMARY KEY ( ID_Institucija ) , FOREIGN KEY (ID_Institucija) REFERENCES Institucija(ID_Institucija));
CREATE TABLE Visoka_Skola (Specijalizacija int, ID_Institucija int,  PRIMARY KEY ( ID_Institucija ) , FOREIGN KEY (ID_Institucija) REFERENCES Institucija(ID_Institucija));
CREATE TABLE Fakultet (BrojGodina int, ID_Institucija int,  PRIMARY KEY ( ID_Institucija ) , FOREIGN KEY (ID_Institucija) REFERENCES Institucija(ID_Institucija));
CREATE TABLE Spominjanje (Vidljivost int, ID_Koris int, Diskriminator int,  PRIMARY KEY ( ID_Koris , Diskriminator ) , FOREIGN KEY (ID_Koris,Diskriminator) REFERENCES Korisnik(ID_Koris,Diskriminator));
CREATE TABLE Ostvaruje (Vreme int, ID_Koris int, Diskriminator int, ID_Poziv int,  PRIMARY KEY ( ID_Koris , Diskriminator , ID_Poziv ) , FOREIGN KEY (ID_Koris,Diskriminator) REFERENCES Korisnik(ID_Koris,Diskriminator) , FOREIGN KEY (ID_Poziv) REFERENCES Poziv(ID_Poziv));
CREATE TABLE Korisnik_Projekat (ID_Koris int, Diskriminator int, ID_Projekat int,  PRIMARY KEY ( ID_Koris , Diskriminator , ID_Projekat ) , FOREIGN KEY (ID_Projekat) REFERENCES Projekat(ID_Projekat) , FOREIGN KEY (ID_Koris,Diskriminator) REFERENCES Korisnik(ID_Koris,Diskriminator));
CREATE TABLE Korisnik_Dozvola (ID_Koris int, Diskriminator int, ID_Dozvola int,  PRIMARY KEY ( ID_Koris , Diskriminator , ID_Dozvola ) , FOREIGN KEY (ID_Koris,Diskriminator) REFERENCES Korisnik(ID_Koris,Diskriminator) , FOREIGN KEY (ID_Dozvola) REFERENCES Dozvola(ID_Dozvola));
CREATE TABLE Korisnik_Server (ID_Koris int, Diskriminator int, ID_Server int,  PRIMARY KEY ( ID_Koris , Diskriminator , ID_Server ) , FOREIGN KEY (ID_Koris,Diskriminator) REFERENCES Korisnik(ID_Koris,Diskriminator) , FOREIGN KEY (ID_Server) REFERENCES Server(ID_Server));
CREATE TABLE Pozivnica_Server (ID_Koris int, Diskriminator int, ID_poziv int,  PRIMARY KEY ( ID_Koris , Diskriminator , ID_poziv ) , FOREIGN KEY (ID_Koris,Diskriminator) REFERENCES Korisnik(ID_Koris,Diskriminator) , FOREIGN KEY (ID_poziv) REFERENCES Pozivnica(ID_poziv));
CREATE TABLE Korisnik_Pozivnica (ID_Koris int, Diskriminator int, ID_Poziv int,  PRIMARY KEY ( ID_Koris , Diskriminator , ID_Poziv ) , FOREIGN KEY (ID_Koris,Diskriminator) REFERENCES Korisnik(ID_Koris,Diskriminator) , FOREIGN KEY (ID_Poziv) REFERENCES Pozivnica(ID_Poziv));
CREATE TABLE Poruka (ID_Poruka int, Vrednost int, BrojKaraktera int, URL int, VremeSlanja int, ID_Koris int, Diskriminator int, ID_Kanal int,  PRIMARY KEY ( ID_Poruka ) , FOREIGN KEY (ID_Koris,Diskriminator) REFERENCES Korisnik(ID_Koris,Diskriminator) , FOREIGN KEY (ID_Kanal) REFERENCES _tekst(ID_Kanal));
CREATE TABLE Datoteka (ID_Dat int, Naziv int, Velicina int, Tip int, ID_Koris int, Diskriminator int, ID_Poruka int,  PRIMARY KEY ( ID_Dat ) , FOREIGN KEY (ID_Koris,Diskriminator) REFERENCES Korisnik(ID_Koris,Diskriminator) , FOREIGN KEY (ID_Poruka) REFERENCES Poruka(ID_Poruka));
CREATE TABLE Reakcija (Emoji int, Zakljucana int, ID_Poruka int, ID_Koris int, Diskriminator int,  PRIMARY KEY ( ID_Koris , Diskriminator ) , FOREIGN KEY (ID_Poruka) REFERENCES Poruka(ID_Poruka) , FOREIGN KEY (ID_Koris,Diskriminator) REFERENCES Korisnik(ID_Koris,Diskriminator));
CREATE TABLE Spominjanje_Poruka (ID_Koris int, Diskriminator int, ID_Poruka int,  PRIMARY KEY ( ID_Koris , Diskriminator , ID_Poruka ) , FOREIGN KEY (ID_Koris,Diskriminator) REFERENCES Korisnik(ID_Koris,Diskriminator) , FOREIGN KEY (ID_Poruka) REFERENCES Poruka(ID_Poruka));
